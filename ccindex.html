                        <ol>

                            <li><strong>Rewards Cards:</strong> These cards let you accumulate points that can be redeemed for a variety of goods, vouchers, or services.                             <li><strong>Annual Fees and Interest Rates:</strong> Consider the card's annual fee, which may be waived for the first year, and the interest rates if you plan to carry a balance.</li>

    <script>

        function filterCards() {

            const filter = document.getElementById('provider-filter').value.toLowerCase();

            const cards = document.querySelectorAll('.card');

 

            cards.forEach(card => {

                const provider = card.getAttribute('data-provider').toLowerCase();

                if (filter === 'all' || provider === filter) {

                    card.style.display = 'block';

                } else {

                    card.style.display = 'none';

                }

            });

        }

 

        // Dynamically ensure the parent container is full width on the live page

        window.addEventListener('load', function() {

            const section = document.querySelector('.credit-section');

            if (section) {

                const parent = section.closest('.container, .row, .section, .cf-section, .cf-row, .cf-container');

                if (parent) {

                    parent.style.width = '100%';

                    parent.style.maxWidth = '100%';

                    parent.style.margin = '0';

                    parent.style.padding = '0';

                }

            }

        });

 

        window.addEventListener('resize', function() {

            const section = document.querySelector('.credit-section');
